\begin{frame}{WYSINWYG \textipa{[w\textsci zinw\textsci  g]}}
\begin{itemize}
    \item what-you-see-is-not-what-you-get \pause
    \item display formatting, fonts etc. in control sequences (like html) instead of directly \pause
    \item initially how all digital text editing worked\pause
    \item not to be confused with: \pause
        \begin{itemize}
            \item WYSINWY\textcolor{UniOrange}{W} what-you-see-is-not-what-you-want (Microsoft Word) \pause
            \item WYSINWY\textcolor{UniOrange}{C} what-you-see-is-not-what-you-copy (web exploit) \pause
            \item WY\textcolor{UniOrange}{W}INWYG what-you-want-is-not-what-you-get (general life experience)
        \end{itemize}
\end{itemize}
\end{frame}
